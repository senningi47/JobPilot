> **适用于智能体工作者：** 必须使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现本计划。步骤使用 `- [ ]` 语法进行进度追踪。

**目标：** 实现 JobPilot MVP 的 5 个 P0 垂直切片功能：用户认证、AI 对话、岗位探索、公司情报、简历分析。

**架构：** 三服务 monorepo — Next.js 14 前端 + Spring Boot 3.x 后端 + Python FastAPI AI 服务。后端通过同步 HTTP 调用 AI 服务。所有 LLM 功能使用 Mock 实现，接口格式与真实 API 一致，便于后续无缝切换。

**技术栈：** Next.js 14, shadcn/ui, Tailwind CSS, TypeScript, Zustand, ECharts, Spring Boot 3.3, Java 17, Spring Security, JPA, MySQL 8, Redis 7, FastAPI, Pydantic, Anthropic SDK (预留)

```java
// backend/src/main/java/com/jobpilot/util/JwtUtil.java
package com.jobpilot.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import javax.crypto.SecretKey;
import java.nio.charset.StandardCharsets;
import java.util.Date;

@Component
public class JwtUtil {

    @Value("${app.jwt.secret}")
    private String secret;

    @Value("${app.jwt.expiration}")
    private long expiration;

    private SecretKey getKey() {
        return Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
    }

    public String generateToken(Long userId, String username, String email) {
        return Jwts.builder()
                .claim("userId", userId)
                .claim("username", username)
                .claim("email", email)
                .issuedAt(new Date())
                .expiration(new Date(System.currentTimeMillis() + expiration * 1000))
                .signWith(getKey())
                .compact();
    }

    public Claims parseToken(String token) {
        return Jwts.parser()
                .verifyWith(getKey())
                .build()
                .parseSignedClaims(token)
                .getPayload();
    }

    public boolean isTokenValid(String token) {
        try {
            Claims claims = parseToken(token);
            return claims.getExpiration().after(new Date());
        } catch (Exception e) {
            return false;
        }
    }
}
```

```java
// backend/src/main/java/com/jobpilot/service/impl/AuthServiceImpl.java
package com.jobpilot.service.impl;

import com.jobpilot.dto.auth.AuthResponse;
import com.jobpilot.dto.auth.LoginRequest;
import com.jobpilot.dto.auth.RegisterRequest;
import com.jobpilot.entity.UserEntity;
import com.jobpilot.exception.BusinessException;
import com.jobpilot.repository.UserRepository;
import com.jobpilot.service.AuthService;
import com.jobpilot.util.JwtUtil;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;

@Service
public class AuthServiceImpl implements AuthService {

    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    private final JwtUtil jwtUtil;

    public AuthServiceImpl(UserRepository userRepository,
                           PasswordEncoder passwordEncoder,
                           JwtUtil jwtUtil) {
        this.userRepository = userRepository;
        this.passwordEncoder = passwordEncoder;
        this.jwtUtil = jwtUtil;
    }

    @Override
    public AuthResponse register(RegisterRequest req) {
        if (userRepository.existsByEmail(req.getEmail())) {
            throw BusinessException.userExists();
        }
        if (userRepository.existsByUsername(req.getUsername())) {
            throw BusinessException.userExists();
        }

        UserEntity user = new UserEntity();
        user.setUsername(req.getUsername());
        user.setEmail(req.getEmail());
        user.setPasswordHash(passwordEncoder.encode(req.getPassword()));
        user.setMajor(req.getMajor());
        user.setGraduationYear(req.getGraduationYear());
        user = userRepository.save(user);

        String token = jwtUtil.generateToken(user.getId(), user.getUsername(), user.getEmail());
        return new AuthResponse(token,
                new AuthResponse.UserInfo(user.getId(), user.getUsername(), user.getEmail(), user.getMajor()));
    }

    @Override
    public AuthResponse login(LoginRequest req) {
        UserEntity user = userRepository.findByEmail(req.getEmail())
                .orElseThrow(() -> new BusinessException(1001, "邮箱或密码错误"));

        if (!passwordEncoder.matches(req.getPassword(), user.getPasswordHash())) {
            throw new BusinessException(1001, "邮箱或密码错误");
        }

        String token = jwtUtil.generateToken(user.getId(), user.getUsername(), user.getEmail());
        return new AuthResponse(token,
                new AuthResponse.UserInfo(user.getId(), user.getUsername(), user.getEmail(), user.getMajor()));
    }

    @Override
    public AuthResponse.UserInfo getCurrentUser(Long userId) {
        UserEntity user = userRepository.findById(userId)
                .orElseThrow(() -> BusinessException.notFound(2001, "用户不存在"));
        return new AuthResponse.UserInfo(user.getId(), user.getUsername(), user.getEmail(), user.getMajor());
    }
}
```

```java
package com.jobpilot.filter;

import com.jobpilot.util.JwtUtil;
import io.jsonwebtoken.Claims;
import jakarta.servlet.FilterChain;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.stereotype.Component;
import org.springframework.web.filter.OncePerRequestFilter;

import java.io.IOException;
import java.util.Collections;

@Component
public class JwtAuthFilter extends OncePerRequestFilter {

    private final JwtUtil jwtUtil;

    public JwtAuthFilter(JwtUtil jwtUtil) {
        this.jwtUtil = jwtUtil;
    }

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain chain) throws ServletException, IOException {
        String header = request.getHeader("Authorization");

        if (header != null && header.startsWith("Bearer ")) {
            String token = header.substring(7);
            if (jwtUtil.isTokenValid(token)) {
                Claims claims = jwtUtil.parseToken(token);
                Long userId = claims.get("userId", Long.class);
                UsernamePasswordAuthenticationToken auth =
                        new UsernamePasswordAuthenticationToken(userId, null, Collections.emptyList());
                SecurityContextHolder.getContext().setAuthentication(auth);
            }
        }

        chain.doFilter(request, response);
    }
}
```

```typescript
// frontend/src/lib/store/authStore.ts
import { create } from "zustand";
import apiClient from "@/lib/api";

interface User {
  id: number;
  username: string;
  email: string;
  major: string | null;
}

interface AuthState {
  token: string | null;
  user: User | null;
  login: (email: string, password: string) => Promise<void>;
  register: (data: {
    username: string;
    email: string;
    password: string;
    major?: string;
    graduationYear?: number;
  }) => Promise<void>;
  logout: () => void;
  loadFromStorage: () => void;
}

export const useAuthStore = create<AuthState>((set) => ({
  token: null,
  user: null,

  login: async (email, password) => {
    const res = await apiClient.post("/v1/auth/login", { email, password });
    const { token, user } = res.data.data;
    localStorage.setItem("token", token);
    localStorage.setItem("user", JSON.stringify(user));
    set({ token, user });
  },

  register: async (data) => {
    const res = await apiClient.post("/v1/auth/register", data);
    const { token, user } = res.data.data;
    localStorage.setItem("token", token);
    localStorage.setItem("user", JSON.stringify(user));
    set({ token, user });
  },

  logout: () => {
    localStorage.removeItem("token");
    localStorage.removeItem("user");
    set({ token: null, user: null });
  },

  loadFromStorage: () => {
    const token = localStorage.getItem("token");
    const userStr = localStorage.getItem("user");
    if (token && userStr) {
      set({ token, user: JSON.parse(userStr) });
    }
  },
}));
```

```python
import json
import os
import random
from app.schemas.chat import ChatResponse, ChatData

MOCK_DATA_PATH = os.path.join(os.path.dirname(__file__), "mock_data", "chat_responses.json")

INTENT_KEYWORDS = {
    "job_search": ["岗位", "工作", "实习", "职位", "方向"],
    "company_info": ["公司", "字节", "腾讯", "阿里", "美团", "百度", "华为"],
    "resume_help": ["简历", "简历分析", "简历优化", "帮我改简历"],
    "salary": ["薪资", "工资", "钱", "薪水", "待遇"],
}


def load_mock_data() -> dict:
    with open(MOCK_DATA_PATH, "r", encoding="utf-8") as f:
        return json.load(f)


def detect_intent(message: str) -> str:
    for intent, keywords in INTENT_KEYWORDS.items():
        for kw in keywords:
            if kw in message:
                return intent
    return "general"


def get_chat_response(message: str) -> ChatResponse:
    mock_data = load_mock_data()
    intent = detect_intent(message)
    intent_data = mock_data.get(intent, mock_data["general"])

    # 按概率返回不同场景：error 5%, empty 10%, normal 85%
    roll = random.random()
    if roll < 0.05:
        return ChatResponse(
            status="error",
            data=None,
            message=intent_data["error"]["message"],
        )
    elif roll < 0.15:
        return ChatResponse(
            status="empty",
            data=None,
            message=intent_data["empty"]["message"],
        )
    else:
        normal = intent_data["normal"]
        return ChatResponse(
            status="success",
            data=ChatData(**normal["data"]),
            model_used="mock",
            latency_ms=random.randint(100, 500),
```

```java
// backend/src/main/java/com/jobpilot/service/impl/ChatServiceImpl.java
package com.jobpilot.service.impl;

import com.jobpilot.dto.PageResponse;
import com.jobpilot.dto.chat.*;
import com.jobpilot.entity.ChatMessageEntity;
import com.jobpilot.entity.ChatSessionEntity;
import com.jobpilot.exception.BusinessException;
import com.jobpilot.repository.ChatMessageRepository;
import com.jobpilot.repository.ChatSessionRepository;
import com.jobpilot.service.ChatService;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.time.Duration;
import java.util.*;
import java.util.stream.Collectors;

@Service
public class ChatServiceImpl implements ChatService {

    private static final Logger log = LoggerFactory.getLogger(ChatServiceImpl.class);
    private static final String REDIS_KEY_PREFIX = "chat:session:";
    private static final int REDIS_MAX_MESSAGES = 20;

    private final ChatSessionRepository sessionRepository;
    private final ChatMessageRepository messageRepository;
    private final RedisTemplate<String, Object> redisTemplate;
    private final RestTemplate restTemplate;
    private final ObjectMapper objectMapper;

    @Value("${app.ai-service.url}")
    private String aiServiceUrl;

    public ChatServiceImpl(ChatSessionRepository sessionRepository,
                           ChatMessageRepository messageRepository,
                           RedisTemplate<String, Object> redisTemplate,
                           RestTemplate restTemplate,
                           ObjectMapper objectMapper) {
        this.sessionRepository = sessionRepository;
        this.messageRepository = messageRepository;
        this.redisTemplate = redisTemplate;
        this.restTemplate = restTemplate;
        this.objectMapper = objectMapper;
    }

    @Override
    public ChatSessionDto createSession(Long userId) {
        ChatSessionEntity session = new ChatSessionEntity();
        session.setSessionId(UUID.randomUUID().toString());
        session.setUserId(userId);
        session.setTitle("新对话");
        session = sessionRepository.save(session);
        return toSessionDto(session);
    }

    @Override
    public PageResponse<ChatSessionDto> listSessions(Long userId, int page, int pageSize) {
        Page<ChatSessionEntity> result = sessionRepository
                .findByUserIdOrderByUpdatedAtDesc(userId, PageRequest.of(page, pageSize));
        List<ChatSessionDto> list = result.getContent().stream()
                .map(this::toSessionDto)
                .collect(Collectors.toList());
        return new PageResponse<>(list, result.getTotalElements(), page, pageSize);
    }

    @Override
    public List<ChatMessageDto> getMessages(Long userId, String sessionId) {
        // 先尝试 Redis
        String redisKey = REDIS_KEY_PREFIX + sessionId + ":messages";
        List<Object> cached = redisTemplate.opsForList().range(redisKey, 0, -1);

        if (cached != null && !cached.isEmpty()) {
            return cached.stream()
                    .map(obj -> objectMapper.convertValue(obj, ChatMessageDto.class))
                    .collect(Collectors.toList());
        }

        // Redis 未命中，从 MySQL 加载最近 20 条并预热缓存
        List<ChatMessageEntity> messages = messageRepository
                .findTop20BySessionIdOrderByCreatedAtDesc(sessionId);
        Collections.reverse(messages);

        List<ChatMessageDto> dtos = messages.stream()
                .map(this::toMessageDto)
                .collect(Collectors.toList());

        // 预热 Redis
        if (!dtos.isEmpty()) {
            redisTemplate.opsForList().rightPushAll(redisKey, dtos.toArray());
            redisTemplate.expire(redisKey, Duration.ofDays(7));
        }

        return dtos;
    }

    @Override
    public ChatMessageDto sendMessage(Long userId, String sessionId, String content) {
        // 1. 保存用户消息到 MySQL
        ChatMessageEntity userMsg = new ChatMessageEntity();
        userMsg.setSessionId(sessionId);
        userMsg.setUserId(userId);
        userMsg.setRole(ChatMessageEntity.MessageRole.user);
        userMsg.setContent(content);
        messageRepository.save(userMsg);

        // 2. LPUSH 到 Redis
        String redisKey = REDIS_KEY_PREFIX + sessionId + ":messages";
        ChatMessageDto userDto = toMessageDto(userMsg);
        redisTemplate.opsForList().leftPush(redisKey, userDto);
        redisTemplate.opsForList().trim(redisKey, 0, REDIS_MAX_MESSAGES - 1);
        redisTemplate.expire(redisKey, Duration.ofDays(7));

        // 3. 调用 AI 服务
        Map<String, Object> aiRequest = new HashMap<>();
        aiRequest.put("session_id", sessionId);
        aiRequest.put("user_id", userId.toString());
        aiRequest.put("message", content);

        Map<String, Object> aiResponse;
        try {
            aiResponse = restTemplate.postForObject(
                    aiServiceUrl + "/chat/send", aiRequest, Map.class);
        } catch (Exception e) {
            log.error("AI 服务调用失败", e);
            throw BusinessException.aiServiceUnavailable();
        }

        // 4. 解析 AI 响应
        String aiContent;
        String intent = null;
        String modelUsed = "mock";
        if ("success".equals(aiResponse.get("status"))) {
            Map<String, Object> data = (Map<String, Object>) aiResponse.get("data");
            aiContent = (String) data.get("content");
            intent = (String) data.get("intent");
        } else {
            aiContent = (String) aiResponse.get("message");
        }
        if (aiResponse.containsKey("model_used")) {
            modelUsed = (String) aiResponse.get("model_used");
        }

        // 5. 保存 AI 响应到 MySQL
        ChatMessageEntity aiMsg = new ChatMessageEntity();
        aiMsg.setSessionId(sessionId);
        aiMsg.setUserId(userId);
        aiMsg.setRole(ChatMessageEntity.MessageRole.assistant);
        aiMsg.setContent(aiContent);
        aiMsg.setIntent(intent);
        aiMsg.setModelUsed(modelUsed);
        messageRepository.save(aiMsg);

        // 6. LPUSH AI 响应到 Redis
        ChatMessageDto aiDto = toMessageDto(aiMsg);
        redisTemplate.opsForList().leftPush(redisKey, aiDto);
        redisTemplate.opsForList().trim(redisKey, 0, REDIS_MAX_MESSAGES - 1);

        // 7. 更新会话消息计数
        sessionRepository.findByUserIdOrderByUpdatedAtDesc(userId, PageRequest.of(0, 1))
                .stream()
                .filter(s -> s.getSessionId().equals(sessionId))
                .findFirst()
                .ifPresent(s -> {
                    s.setMessageCount(s.getMessageCount() + 2);
                    sessionRepository.save(s);
                });

        return aiDto;
    }

    private ChatSessionDto toSessionDto(ChatSessionEntity entity) {
        ChatSessionDto dto = new ChatSessionDto();
        dto.setId(entity.getId());
        dto.setSessionId(entity.getSessionId());
        dto.setTitle(entity.getTitle());
        dto.setMessageCount(entity.getMessageCount());
        dto.setCreatedAt(entity.getCreatedAt());
        dto.setUpdatedAt(entity.getUpdatedAt());
        return dto;
    }

    private ChatMessageDto toMessageDto(ChatMessageEntity entity) {
        ChatMessageDto dto = new ChatMessageDto();
        dto.setId(entity.getId());
        dto.setRole(entity.getRole().name());
        dto.setContent(entity.getContent());
        dto.setIntent(entity.getIntent());
        dto.setModelUsed(entity.getModelUsed());
        dto.setTokenCount(entity.getTokenCount());
        dto.setCreatedAt(entity.getCreatedAt());
        return dto;
    }
}
```

```typescript
  id: number;
  sessionId: string;
  title: string;
  messageCount: number;
  createdAt: string;
  updatedAt: string;
}

interface ChatState {
  sessions: ChatSession[];
  currentSessionId: string | null;
  messages: ChatMessage[];
  loading: boolean;
  sending: boolean;
  fetchSessions: () => Promise<void>;
  createSession: () => Promise<string>;
  selectSession: (sessionId: string) => Promise<void>;
  sendMessage: (content: string) => Promise<void>;
}

export const useChatStore = create<ChatState>((set, get) => ({
  sessions: [],
  currentSessionId: null,
  messages: [],
  loading: false,
  sending: false,

  fetchSessions: async () => {
    const res = await apiClient.get("/v1/chat/sessions");
    set({ sessions: res.data.data.list });
  },

  createSession: async () => {
    const res = await apiClient.post("/v1/chat/sessions");
    const session = res.data.data;
    set((s) => ({ sessions: [session, ...s.sessions], currentSessionId: session.sessionId, messages: [] }));
    return session.sessionId;
  },

  selectSession: async (sessionId) => {
    set({ loading: true, currentSessionId: sessionId });
    const res = await apiClient.get(`/v1/chat/sessions/${sessionId}/messages`);
    set({ messages: res.data.data, loading: false });
  },

  sendMessage: async (content) => {
    const { currentSessionId } = get();
    if (!currentSessionId) return;

    set({ sending: true });
    // 乐观添加用户消息
    const tempUserMsg: ChatMessage = {
      id: Date.now(),
      role: "user",
      content,
      intent: null,
      modelUsed: null,
      createdAt: new Date().toISOString(),
    };
    set((s) => ({ messages: [...s.messages, tempUserMsg] }));

    try {
      const res = await apiClient.post(
        `/v1/chat/sessions/${currentSessionId}/messages`,
        { content }
      );
      const aiMsg: ChatMessage = res.data.data;
      set((s) => ({ messages: [...s.messages, aiMsg], sending: false }));
    } catch {
      set((s) => ({
        messages: [...s.messages, {
          id: Date.now() + 1,
          role: "assistant" as const,
          content: "发送失败，请重试",
          intent: null,
          modelUsed: null,
          createdAt: new Date().toISOString(),
        }],
        sending: false,
      }));
    }
  },
}));
```

```tsx
// frontend/src/components/chat/TypewriterText.tsx
"use client";

import { useState, useEffect } from "react";
import ReactMarkdown from "react-markdown";

interface Props {
  content: string;
  speed?: number;
}

export function TypewriterText({ content, speed = 20 }: Props) {
  const [displayed, setDisplayed] = useState("");
  const [done, setDone] = useState(false);

  useEffect(() => {
    setDisplayed("");
    setDone(false);
    let i = 0;
    const interval = setInterval(() => {
      i++;
      setDisplayed(content.slice(0, i));
      if (i >= content.length) {
        clearInterval(interval);
        setDone(true);
      }
    }, speed);
    return () => clearInterval(interval);
  }, [content, speed]);

  return (
    <div className="prose prose-sm max-w-none">
      <ReactMarkdown>{done ? content : displayed}</ReactMarkdown>
      {!done && <span className="animate-pulse">|</span>}
    </div>
  );
}
```

```python
# ai-service/app/services/job_explore_service.py
import json
import os
import random

DATA_PATH = os.path.join(os.path.dirname(__file__), "mock_data", "job_knowledge_graph.json")


def load_knowledge_graph() -> list[dict]:
    with open(DATA_PATH, "r", encoding="utf-8") as f:
        return json.load(f)


def get_categories() -> list[str]:
    data = load_knowledge_graph()
    return sorted(set(item["category"] for item in data))


def get_majors_by_category(category: str) -> list[dict]:
    data = load_knowledge_graph()
    return [{"major": item["major"], "category": item["category"]}
            for item in data if item["category"] == category]


def get_major_detail(major: str) -> dict | None:
    data = load_knowledge_graph()
    for item in data:
        if item["major"] == major:
            return item
    return None


def search_jobs(query: str) -> list[dict]:
    if not query or len(query.strip()) == 0:
        return []

    query = query[:200]  # 截断到 200 字符
    query_lower = query.lower()
    results = []

    data = load_knowledge_graph()
    for item in data:
        for job in item["primary_jobs"] + item["extended_jobs"]:
            searchable = f"{job['job_title']} {' '.join(job.get('tags', []))} {job.get('description', '')}"
            if query_lower in searchable.lower():
                results.append({
                    "job_title": job["job_title"],
                    "major": item["major"],
                    "category": item["category"],
                    "tags": job.get("tags", []),
                    "description": job.get("description", ""),
                    "confidence": round(random.uniform(0.6, 0.99), 2),
                })

    # 按置信度排序，返回前 5 条
    results.sort(key=lambda x: x["confidence"], reverse=True)
    return results[:5]
```

```python
    if not majors:

    if not detail:
        return {"status": "empty", "data": None, "message": f"未找到专业: {major}"}
    return {"status": "success", "data": detail}


@router.get("/search")

    results = search_jobs(q)
    if not results:
        return {"status": "empty", "data": [], "message": "未找到相关岗位"}
```

```java
package com.jobpilot.service.impl;

import com.jobpilot.exception.BusinessException;
import com.jobpilot.service.JobExploreService;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.util.*;

@Service
public class JobExploreServiceImpl implements JobExploreService {

    private final RestTemplate restTemplate;

    @Value("${app.ai-service.url}")
    private String aiServiceUrl;

    public JobExploreServiceImpl(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    @Override
    public List<String> getCategories() {
        Map<String, Object> resp = restTemplate.getForObject(
                aiServiceUrl + "/jobs/categories", Map.class);
        return extractData(resp);
    }

    @Override
    public List<Map<String, Object>> getMajorsByCategory(String category) {
        Map<String, Object> resp = restTemplate.getForObject(
                aiServiceUrl + "/jobs/categories/" + category + "/majors", Map.class);
        return extractData(resp);
    }

    @Override
    public Map<String, Object> getMajorDetail(String major) {
        Map<String, Object> resp = restTemplate.getForObject(
                aiServiceUrl + "/jobs/majors/" + major, Map.class);
        return extractData(resp);
    }

    @Override
    public List<Map<String, Object>> searchJobs(String query) {
        Map<String, Object> resp = restTemplate.getForObject(
                aiServiceUrl + "/jobs/search?q=" + query, Map.class);
        return extractData(resp);
    }

    @SuppressWarnings("unchecked")
    private <T> T extractData(Map<String, Object> resp) {
        if (resp == null || !"success".equals(resp.get("status"))) {
            String msg = resp != null ? (String) resp.get("message") : "AI 服务无响应";
            throw BusinessException.notFound(2001, msg);
        }
        return (T) resp.get("data");
    }
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { Input } from "@/components/ui/input";
import { Card } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import apiClient from "@/lib/api";

interface SearchResult {
  job_title: string;
  major: string;
  tags: string[];
  confidence: number;
}

interface Props {
  onSelect?: (result: SearchResult) => void;
}

export function SearchBar({ onSelect }: Props) {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<SearchResult[]>([]);
  const [open, setOpen] = useState(false);

  const search = useCallback(async (q: string) => {
    if (!q.trim()) { setResults([]); return; }
    try {
      const res = await apiClient.get(`/v1/jobs/search?q=${encodeURIComponent(q)}`);
      setResults(res.data.data || []);
      setOpen(true);
    } catch { setResults([]); }
  }, []);

  useEffect(() => {
    const timer = setTimeout(() => search(query), 300);
    return () => clearTimeout(timer);
  }, [query, search]);

  return (
    <div className="relative w-full max-w-xl mx-auto">
      <Input
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="搜索岗位、技能或关键词..."
        className="w-full"
        onFocus={() => results.length > 0 && setOpen(true)}
        onBlur={() => setTimeout(() => setOpen(false), 200)}
      />
      {open && results.length > 0 && (
        <Card className="absolute z-10 w-full mt-1 p-2 shadow-lg">
          {results.map((r, i) => (
            <div
              key={i}
              className="flex items-center justify-between p-2 hover:bg-gray-50 rounded cursor-pointer"
              onClick={() => { onSelect?.(r); setOpen(false); }}
            >
              <div>
                <span className="font-medium text-sm">{r.job_title}</span>
                <span className="text-xs text-gray-400 ml-2">{r.major}</span>
              </div>
              <div className="flex gap-1">
                {r.tags.slice(0, 3).map((t) => (
                  <Badge key={t} variant="secondary" className="text-xs">{t}</Badge>
                ))}
              </div>
            </div>
          ))}
        </Card>
      )}
    </div>
  );
}
```

```python
import json
import os
import random

DATA_PATH = os.path.join(os.path.dirname(__file__), "mock_data", "company_data.json")


def load_company_data() -> list[dict]:
    with open(DATA_PATH, "r", encoding="utf-8") as f:
        return json.load(f)


def get_company(name: str) -> dict | None:
    data = load_company_data()
    for company in data:
        if company["name"] == name or company["name_en"].lower() == name.lower():
            return company
    return None


def search_companies(query: str) -> list[dict]:
    if not query:
        return []
    data = load_company_data()
    return [
        {"name": c["name"], "name_en": c["name_en"], "industry": c["basic_info"]["industry"]}
        for c in data
        if query in c["name"] or query.lower() in c["name_en"].lower()
    ]
```

```java
// backend/src/main/java/com/jobpilot/service/impl/CompanyServiceImpl.java
package com.jobpilot.service.impl;

import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.jobpilot.entity.CompanyCacheEntity;
import com.jobpilot.exception.BusinessException;
import com.jobpilot.repository.CompanyCacheRepository;
import com.jobpilot.service.CompanyService;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;

import java.time.Duration;
import java.time.LocalDateTime;
import java.util.*;

@Service
public class CompanyServiceImpl implements CompanyService {

    private static final Logger log = LoggerFactory.getLogger(CompanyServiceImpl.class);
    private static final String REDIS_PREFIX = "company:intel:";

    private final CompanyCacheRepository cacheRepository;
    private final RedisTemplate<String, Object> redisTemplate;
    private final RestTemplate restTemplate;
    private final ObjectMapper objectMapper;

    @Value("${app.ai-service.url}")
    private String aiServiceUrl;

    public CompanyServiceImpl(CompanyCacheRepository cacheRepository,
                              RedisTemplate<String, Object> redisTemplate,
                              RestTemplate restTemplate,
                              ObjectMapper objectMapper) {
        this.cacheRepository = cacheRepository;
        this.redisTemplate = redisTemplate;
        this.restTemplate = restTemplate;
        this.objectMapper = objectMapper;
    }

    @Override
    public List<Map<String, Object>> searchCompanies(String query) {
        Map<String, Object> resp = restTemplate.getForObject(
                aiServiceUrl + "/companies/search?q=" + query, Map.class);
        if (resp == null || !"success".equals(resp.get("status"))) {
            return Collections.emptyList();
        }
        return (List<Map<String, Object>>) resp.get("data");
    }

    @Override
    public Map<String, Object> getCompanyIntel(String name) {
        // 1. Redis 缓存
        String redisKey = REDIS_PREFIX + name;
        Object cached = redisTemplate.opsForValue().get(redisKey);
        if (cached != null) {
            return objectMapper.convertValue(cached, new TypeReference<>() {});
        }

        // 2. MySQL 缓存
        Optional<CompanyCacheEntity> dbCache = cacheRepository.findByCompanyName(name);
        if (dbCache.isPresent() && dbCache.get().getExpiresAt().isAfter(LocalDateTime.now())) {
            Map<String, Object> data = buildFromEntity(dbCache.get());
            redisTemplate.opsForValue().set(redisKey, data, Duration.ofHours(24));
            return data;
        }

        // 3. 调用 AI 服务
        return fetchFromAiAndCache(name, redisKey);
    }

    @Override
    public Map<String, Object> refreshCompany(String name) {
        String redisKey = REDIS_PREFIX + name;
        redisTemplate.delete(redisKey);
        cacheRepository.findByCompanyName(name).ifPresent(cacheRepository::delete);
        return fetchFromAiAndCache(name, redisKey);
    }

    @SuppressWarnings("unchecked")
    private Map<String, Object> fetchFromAiAndCache(String name, String redisKey) {
        Map<String, Object> resp;
        try {
            resp = restTemplate.getForObject(aiServiceUrl + "/companies/" + name, Map.class);
        } catch (Exception e) {
            log.error("AI 服务调用失败", e);
            throw BusinessException.aiServiceUnavailable();
        }

        if (resp == null || !"success".equals(resp.get("status"))) {
            String msg = resp != null ? (String) resp.get("message") : "未找到公司信息";
            throw BusinessException.notFound(2001, msg);
        }

        Map<String, Object> data = (Map<String, Object>) resp.get("data");

        // 写入 MySQL
        try {
            CompanyCacheEntity entity = new CompanyCacheEntity();
            entity.setCompanyName(name);
            entity.setCompanyNameEn((String) data.get("name_en"));
            entity.setBasicInfo(objectMapper.writeValueAsString(data.get("basic_info")));
            entity.setSalaryData(objectMapper.writeValueAsString(data.get("salary_data")));
            entity.setReviewSummary(objectMapper.writeValueAsString(data.get("review_summary")));
            entity.setTimeline(objectMapper.writeValueAsString(data.get("timeline")));
            entity.setDataSource("mock");
            entity.setLastUpdated(LocalDateTime.now());
            entity.setExpiresAt(LocalDateTime.now().plusHours(24));
            cacheRepository.save(entity);
        } catch (Exception e) {
            log.warn("缓存写入失败", e);
        }

        // 写入 Redis
        redisTemplate.opsForValue().set(redisKey, data, Duration.ofHours(24));

        return data;
    }

    private Map<String, Object> buildFromEntity(CompanyCacheEntity entity) {
        Map<String, Object> data = new LinkedHashMap<>();
        data.put("name", entity.getCompanyName());
        data.put("name_en", entity.getCompanyNameEn());
        try {
            data.put("basic_info", objectMapper.readValue(entity.getBasicInfo(), Map.class));
            data.put("salary_data", objectMapper.readValue(entity.getSalaryData(), Map.class));
            data.put("review_summary", objectMapper.readValue(entity.getReviewSummary(), Map.class));
            data.put("timeline", objectMapper.readValue(entity.getTimeline(), List.class));
        } catch (Exception e) {
            log.warn("反序列化失败", e);
        }
        return data;
    }
}
```

```tsx
interface CompanySuggestion {
  name: string;

export function CompanySearch() {
  const [query, setQuery] = useState("");
  const [suggestions, setSuggestions] = useState<CompanySuggestion[]>([]);
  const [open, setOpen] = useState(false);
  const router = useRouter();

  const search = useCallback(async (q: string) => {
    if (!q.trim()) { setSuggestions([]); return; }
    try {
      const res = await apiClient.get(`/v1/companies/search?q=${encodeURIComponent(q)}`);
      setSuggestions(res.data.data || []);
      setOpen(true);
    } catch { setSuggestions([]); }
  }, []);

  useEffect(() => {
    const timer = setTimeout(() => search(query), 300);
    return () => clearTimeout(timer);
  }, [query, search]);

  return (
    <div className="relative w-full max-w-xl mx-auto">
      <Input
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="搜索公司名称..."

        onBlur={() => setTimeout(() => setOpen(false), 200)}
      />
      {open && suggestions.length > 0 && (
        <Card className="absolute z-10 w-full mt-1 p-2 shadow-lg">
          {suggestions.map((s) => (
            <div
```

```python
    async def save(self, user_id: str, filename: str, content: bytes) -> str:
        user_dir = os.path.join(self.BASE_DIR, str(user_id))
        os.makedirs(user_dir, exist_ok=True)
        timestamp = int(time.time())
        safe_name = f"{timestamp}_{filename}"
        path = os.path.join(user_dir, safe_name)
        with open(path, "wb") as f:
            f.write(content)
        return path

    async def read(self, path: str) -> bytes:
        with open(path, "rb") as f:
            return f.read()

    async def delete(self, path: str) -> bool:
        if os.path.exists(path):
            os.remove(path)
            return True
        return False
```

```java
// backend/src/main/java/com/jobpilot/service/impl/ResumeServiceImpl.java
package com.jobpilot.service.impl;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.jobpilot.entity.ResumeEntity;
import com.jobpilot.exception.BusinessException;
import com.jobpilot.repository.ResumeRepository;
import com.jobpilot.service.ResumeService;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.core.io.ByteArrayResource;
import org.springframework.http.HttpEntity;
import org.springframework.http.HttpHeaders;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Service;
import org.springframework.util.LinkedMultiValueMap;
import org.springframework.util.MultiValueMap;
import org.springframework.web.client.RestTemplate;
import org.springframework.web.multipart.MultipartFile;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.time.Instant;
import java.util.*;

@Service
public class ResumeServiceImpl implements ResumeService {

    private static final Logger log = LoggerFactory.getLogger(ResumeServiceImpl.class);

    private final ResumeRepository resumeRepository;
    private final RestTemplate restTemplate;
    private final ObjectMapper objectMapper;

    @Value("${app.ai-service.url}")
    private String aiServiceUrl;

    public ResumeServiceImpl(ResumeRepository resumeRepository,
                             RestTemplate restTemplate,
                             ObjectMapper objectMapper) {
        this.resumeRepository = resumeRepository;
        this.restTemplate = restTemplate;
        this.objectMapper = objectMapper;
    }

    @Override
    @SuppressWarnings("unchecked")
    public Map<String, Object> uploadResume(Long userId, MultipartFile file) {
        String filename = file.getOriginalFilename();
        if (filename == null || (!filename.endsWith(".pdf") && !filename.endsWith(".docx"))) {
            throw new BusinessException(2002, "不支持的文件格式，请上传 PDF 或 DOCX");
        }
        if (file.getSize() > 10 * 1024 * 1024) {
            throw new BusinessException(2002, "文件大小不能超过 10MB");
        }

        // 保存文件到本地
        String filePath;
        try {
            String timestamp = String.valueOf(Instant.now().getEpochSecond());
            String safeName = timestamp + "_" + filename;
            Path dir = Paths.get("/uploads", String.valueOf(userId));
            Files.createDirectories(dir);
            Path fullPath = dir.resolve(safeName);
            file.transferTo(fullPath.toFile());
            filePath = fullPath.toString();
        } catch (IOException e) {
            log.error("文件保存失败", e);
            throw new BusinessException(5001, "文件保存失败");
        }

        // 调用 AI 服务解析
        Map<String, Object> aiResp;
        try {
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.MULTIPART_FORM_DATA);
            MultiValueMap<String, Object> body = new LinkedMultiValueMap<>();
            body.add("file", new ByteArrayResource(file.getBytes()) {
                @Override
                public String getFilename() { return filename; }
            });
            HttpEntity<MultiValueMap<String, Object>> requestEntity = new HttpEntity<>(body, headers);
            aiResp = restTemplate.postForObject(
                    aiServiceUrl + "/resume/upload", requestEntity, Map.class);
        } catch (Exception e) {
            log.error("AI 服务调用失败", e);
            throw BusinessException.aiServiceUnavailable();
        }

        if (aiResp == null || !"success".equals(aiResp.get("status"))) {
            String msg = aiResp != null ? (String) aiResp.get("message") : "简历解析失败";
            throw new BusinessException(2002, msg);
        }

        // 保存到 MySQL
        ResumeEntity entity = new ResumeEntity();
        entity.setUserId(userId);
        try {
            entity.setStructuredData(objectMapper.writeValueAsString(aiResp.get("data")));
        } catch (Exception e) {
            log.warn("序列化失败", e);
        }
        entity.setRawFileUrl(filePath);
        entity = resumeRepository.save(entity);

        return buildResumeDto(entity);
    }

    @Override
    public List<Map<String, Object>> listResumes(Long userId) {
        return resumeRepository.findByUserIdAndIsActiveTrueOrderByCreatedAtDesc(userId)
                .stream().map(this::buildResumeDto).toList();
    }

    @Override
    public Map<String, Object> getResume(Long userId, Long resumeId) {
        ResumeEntity entity = resumeRepository.findById(resumeId)
                .filter(r -> r.getUserId().equals(userId))
                .orElseThrow(() -> BusinessException.notFound(2001, "简历不存在"));
        return buildResumeDto(entity);
    }

    @Override
    @SuppressWarnings("unchecked")
    public Map<String, Object> analyzeResume(Long userId, Long resumeId, String targetPosition) {
        // 确认简历存在
        resumeRepository.findById(resumeId)
                .filter(r -> r.getUserId().equals(userId))
                .orElseThrow(() -> BusinessException.notFound(2001, "简历不存在"));

        // 调用 AI 分析
        try {
            MultiValueMap<String, String> body = new LinkedMultiValueMap<>();
            body.add("target_position", targetPosition);
            HttpEntity<MultiValueMap<String, String>> request = new HttpEntity<>(body);
            Map<String, Object> resp = restTemplate.postForObject(
                    aiServiceUrl + "/resume/analyze", request, Map.class);

            if (resp == null || !"success".equals(resp.get("status"))) {
                String msg = resp != null ? (String) resp.get("message") : "分析失败";
                throw new BusinessException(2002, msg);
            }
            return (Map<String, Object>) resp.get("data");
        } catch (BusinessException e) {
            throw e;
        } catch (Exception e) {
            log.error("AI 分析调用失败", e);
            throw BusinessException.aiServiceUnavailable();
        }
    }

    private Map<String, Object> buildResumeDto(ResumeEntity entity) {
        Map<String, Object> dto = new LinkedHashMap<>();
        dto.put("id", entity.getId());
        dto.put("rawFileUrl", entity.getRawFileUrl());
        dto.put("version", entity.getVersion());
        dto.put("createdAt", entity.getCreatedAt());
        dto.put("updatedAt", entity.getUpdatedAt());
        try {
            if (entity.getStructuredData() != null) {
                dto.put("structuredData", objectMapper.readValue(entity.getStructuredData(), Map.class));
            }
        } catch (Exception e) {
            log.warn("反序列化失败", e);
        }
        return dto;
    }
}
```

```tsx
  const [uploading, setUploading] = useState(false);
  const [dragOver, setDragOver] = useState(false);
  const router = useRouter();

  const handleDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault();
    setDragOver(false);
    const dropped = e.dataTransfer.files[0];
    if (dropped && (dropped.name.endsWith(".pdf") || dropped.name.endsWith(".docx"))) {
      setFile(dropped);
    }
  }, []);

  const handleUpload = async () => {
    if (!file) return;
    setUploading(true);
    const form = new FormData();
    form.append("file", file);
    try {
      const res = await apiClient.post("/v1/resumes/upload", form, {
        headers: { "Content-Type": "multipart/form-data" },
      });
      router.push(`/resume/${res.data.data.id}`);
    } catch (err: any) {
      alert(err.response?.data?.msg || "上传失败");
    } finally {
      setUploading(false);
    }
  };

  return (
    <Card
      className={`border-2 border-dashed p-8 text-center transition-colors ${
        dragOver ? "border-primary-500 bg-primary-50" : "border-gray-300"
      }`}
      onDragOver={(e) => { e.preventDefault(); setDragOver(true); }}
      onDragLeave={() => setDragOver(false)}
      onDrop={handleDrop}
```

```tsx
  const handleAnalyze = async () => {
    const position = resume?.structuredData?.basic_info?.target_position || "后端开发工程师";
    setAnalyzing(true);
    try {
      const res = await apiClient.post(`/v1/resumes/${id}/analyze?targetPosition=${encodeURIComponent(position)}`);
      setScore(res.data.data);
    } catch (err: any) {
      alert(err.response?.data?.msg || "分析失败");
    } finally {
      setAnalyzing(false);
    }
  };

  if (!resume) return <p className="p-8 text-center">加载中...</p>;

  return (
    <main className="container mx-auto py-8 px-4">
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold">简历详情</h1>
        <Button onClick={handleAnalyze} disabled={analyzing}>
          {analyzing ? "分析中..." : "分析匹配度"}
        </Button>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        <ResumePreview data={resume.structuredData} />

        {score && (
          <div className="space-y-4">
            <Card>
              <CardHeader><CardTitle>匹配度分析</CardTitle></CardHeader>
              <CardContent>
                <ScoreRadar dimensions={score.dimensions} overallMatch={score.overall_match} />
              </CardContent>
            </Card>

            <Card>
              <CardHeader><CardTitle>亮点</CardTitle></CardHeader>
              <CardContent>
                <ul className="space-y-1">
                  {score.highlights?.map((h: string, i: number) => (
                    <li key={i} className="text-sm text-green-700">✅ {h}</li>
                  ))}
                </ul>
              </CardContent>
            </Card>

            <Card>
              <CardHeader><CardTitle>待改进</CardTitle></CardHeader>
              <CardContent>
                <ul className="space-y-1">
                  {score.weaknesses?.map((w: string, i: number) => (
                    <li key={i} className="text-sm text-orange-600">⚠ {w}</li>
                  ))}
                </ul>
              </CardContent>
            </Card>

            <Card>
              <CardHeader><CardTitle>改进建议</CardTitle></CardHeader>
              <CardContent>
                <ol className="space-y-1 list-decimal list-inside">
                  {score.suggestions?.map((s: string, i: number) => (
                    <li key={i} className="text-sm">{s}</li>
                  ))}
                </ol>
              </CardContent>
            </Card>
          </div>
        )}
      </div>
    </main>
  );
}
```
